Practical 4: Recommendation System


# Practical 4

# prerequisites for each subject
prerequisites = {
    'math': [],
    'algebra': ['math'],
    'arithmetic': [],
    'physics': ['algebra'],
    'chemistry': ['math'],
    'biology': ['chemistry'],
    'computer_science': ['algebra']
}

# knowledge of each student
knows = {
    'john': ['arithmetic'],
    'jane': ['algebra'],
    'mary': ['math'],
    'jack': ['chemistry']
}

# Learning goals for each student
goals = {
    'john': 'computer_science',
    'jane': 'physics',
    'mary': 'biology',
    'jack': 'computer_science'
}

# function to check if a student can learn a subject
def can_learn(student, subject):
    prerequisites_for_subject = prerequisites.get(subject, [])
    return all(prereq in knows.get(student, []) for prereq in prerequisites_for_subject)

# function to recommend the next step for a student
def recommend_next_step(student):
    goal = goals.get(student)
    if not goal:
        return None

    # check missing prerequisite
    for prerequisite in prerequisites.get(goal, []):
        if prerequisite not in knows.get(student, []):
            return prerequisite

    # if all prerequisites are known
    return goal


print("Recommend next step for John:", recommend_next_step('john'))
print("Recommend next step for Jane:", recommend_next_step('jane'))
print("Recommend next step for Mary:", recommend_next_step('mary'))
print("Recommend next step for Jack:", recommend_next_step('jack'))
