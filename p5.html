Practical 5: A* and AO* Algorithms

A* Algorithm


import heapq

distances = {
    ('a', 'b'): 1,
    ('a', 'c'): 4,
    ('b', 'c'): 2,
    ('b', 'd'): 5,
    ('c', 'd'): 1
}

heuristics = {
    ('a', 'd'): 7,
    ('b', 'd'): 5,
    ('c', 'd'): 2,
    ('d', 'd'): 0
}

def get_neighbors(node):
    neighbors = []
    for (start, end), dist in distances.items():
        if start == node:
            neighbors.append((end, dist))
        elif end == node:
            neighbors.append((start, dist))
    return neighbors

def a_star(start, goal):
    frontier = []
    heapq.heappush(frontier, (heuristics[(start, goal)], 0, [start]))
    visited = set()

    while frontier:
        f_score, g_score, path = heapq.heappop(frontier)
        current = path[-1]

        if current == goal:
            return path, g_score

        if current not in visited:
            visited.add(current)

            for neighbor, step_cost in get_neighbors(current):
                if neighbor not in visited:
                    new_g = g_score + step_cost
                    new_f = new_g + heuristics.get((neighbor, goal), float('inf'))
                    heapq.heappush(frontier, (new_f, new_g, path + [neighbor]))

    return None, float('inf')

path, cost = a_star('a', 'd')
print("A* Path:", path)
print("Cost:", cost)

AO* Algorithm


import heapq

distances = {
    ('a', 'b'): 1,
    ('a', 'c'): 4,
    ('b', 'c'): 2,
    ('b', 'd'): 5,
    ('c', 'd'): 1
}

heuristics = {
    ('a', 'd'): 7,
    ('b', 'd'): 5,
    ('c', 'd'): 2,
    ('d', 'd'): 0
}

def get_neighbors(node):
    neighbors = []
    for (start, end), dist in distances.items():
        if start == node:
            neighbors.append((end, dist))
        elif end == node:
            neighbors.append((start, dist))
    return neighbors

def ao_star(start, goal):
    visited = set()

    def ao_search(current, path, cost):
        if current == goal:
            return path, cost

        visited.add(current)
        best_path = None
        best_cost = float('inf')

        for neighbor, step_cost in get_neighbors(current):
            if neighbor not in visited:
                new_cost = cost + step_cost
                new_path, final_cost = ao_search(
                    neighbor, path + [neighbor], new_cost
                )

                if final_cost < best_cost:
                    best_cost = final_cost
                    best_path = new_path

        visited.remove(current)
        return best_path, best_cost

    return ao_search(start, [start], 0)

path, cost = ao_star('a', 'd')
print("AO* Path:", path)
print("Cost:", cost)
