Dynamic CSP (Excel Integration)


import pandas as pd
from datetime import datetime
import os

doctor={
    "name":"Dr_A",
    "available":["10AM","11AM","12PM","2PM","3PM"],
    "booked":{}
}

EXCEL_FILE="appointments.xlsx"
SHEET_NAME="Appointments"

def initialize_excel():
  if not os.path.exists(EXCEL_FILE):
    row=[]
    for slot in doctor["available"]:
      row.append([
          doctor["name"],
          slot,
          "Available",
          "Available",
          datetime.now().strftime("%Y-%m-%d %H:%M:%S")
      ])

    df = pd.DataFrame(row, columns=[
         "Doctor", "Slot", "Patient", "Status", "Timestamp"
    ])
    df.to_excel(EXCEL_FILE, sheet_name=SHEET_NAME, index=False)

def update_excel(slot, patient_name, status):
  df=pd.read_excel(EXCEL_FILE, sheet_name=SHEET_NAME)
  timestamp=datetime.now().strftime("%Y-%m-%d %H:%M:%S")

  if slot in df["Slot"].values:
    df.loc[df["Slot"] == slot, ["Patient", "Status", "Timestamp"]] = [patient_name, status, timestamp]
  else:
    df.loc[len(df)] = [
        doctor["name"],
        slot,
        patient_name,
        status,
        timestamp
    ]

  df.to_excel(EXCEL_FILE, sheet_name=SHEET_NAME, index=False)

initialize_excel()

while True:
  print("\n--- Doctor Appointment System ---")

  choice = input("Enter your choice: ")

  if choice == "1":
        patient_name = input("Enter patient name: ")

        if slot in doctor["available"]:
            doctor["available"].remove(slot)
            doctor["booked"][slot] = patient_name
