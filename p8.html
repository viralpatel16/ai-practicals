from pyswip import Prolog
import tempfile
import os

prolog = Prolog()

prolog_code_string = r"""
% Facts
father(a, b).
father(a, c).
father(b, d).
father(b, e).
father(c, f).

parent(X, Y) :- father(X, Y).

brother(X, Y) :-
father(F, X),
father(F, Y),
X \= Y.

cousin(X, Y) :-
father(F1, X),
father(F2, Y),
F1 \= F2,
father(G, F1),
father(G, F2).

grandson(X, Y) :-
father(Y, F),
father(F, X).

descendent(X, Y) :-
father(Y, X).
descendent(X, Y) :-
father(Z, X),
descendent(Z, Y).
"""

with tempfile.NamedTemporaryFile(mode='w', delete=False, suffix='.pl') as temp_file:
    temp_file.write(prolog_code_string)
    temp_file_path = temp_file.name

try:
    prolog.consult(temp_file_path)

    print("Executing Prolog queries:")

    if list(prolog.query("brother(d, e).")):
        print('d and e are brothers')

    if list(prolog.query("cousin(d, f).")):
        print('d and f are cousins')

    if list(prolog.query("grandson(d, a).")):
        print('d is a grandson of a')

    if list(prolog.query("descendent(d, a).")):
        print('d is a descendent of a')

finally:
    os.remove(temp_file_path)
