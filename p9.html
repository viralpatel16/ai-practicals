Practical 9: Game Development (Tic-Tac-Toe & Rock-Paper-Scissors)

Tic-Tac-Toe Game


board = [' ']*9

def print_board():
    print()
    print(board[0], '|', board[1], '|', board[2])
    print('--+---+--')
    print(board[3], '|', board[4], '|', board[5])
    print('--+---+--')
    print(board[6], '|', board[7], '|', board[8])
    print()

def check_winner(player):
    win_conditions = [
        [0,1,2], [3,4,5], [6,7,8],
        [0,3,6], [1,4,7], [2,5,8],
        [0,4,8], [2,4,6]
    ]
    for condition in win_conditions:
        if board[condition[0]] == board[condition[1]] == board[condition[2]] == player:
            return True
    return False

def is_draw():
    return ' ' not in board

current_player = 'X'

while True:
    print_board()
    move = int(input(f"Player {current_player}, enter position (1-9): ")) - 1

    if board[move] != ' ':
        print("Position already taken. Try again.")
        continue

    board[move] = current_player

    if check_winner(current_player):
        print_board()
        print(f"Player {current_player} wins!")
        break

    if is_draw():
        print_board()
        print("Game Draw!")
        break

    current_player = 'O' if current_player == 'X' else 'X'

Rock-Paper-Scissors Game


def print_choices():
    print("\nChoices:")
    print("1. Rock")
    print("2. Paper")
    print("3. Scissors\n")

def get_choice(player):
    while True:
        choice = int(input(f"Player {player}, enter your choice (1-3): "))
        if choice in [1, 2, 3]:
            return choice
        else:
            print("Invalid choice. Try again.")

def check_winner(p1, p2):
    if p1 == p2:
        return "Draw"
    elif (p1 == 1 and p2 == 3) or \
         (p1 == 2 and p2 == 1) or \
         (p1 == 3 and p2 == 2):
        return "Player 1"
    else:
        return "Player 2"

choices = {1: "Rock", 2: "Paper", 3: "Scissors"}

print("Rock-Paper-Scissors Game")

print_choices()

p1_choice = get_choice(1)
p2_choice = get_choice(2)

print(f"\nPlayer 1 chose: {choices[p1_choice]}")
print(f"Player 2 chose: {choices[p2_choice]}")

result = check_winner(p1_choice, p2_choice)

if result == "Draw":
    print("Game Draw!")
else:
    print(f"{result} wins!")
