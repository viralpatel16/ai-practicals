Simple NLP


nouns = ["dog", "cat", "student", "teacher", "exam"]
verbs = ["eat", "run", "study", "teach", "write"]
positive_words = ["good", "happy", "excellent", "love"]
negative_words = ["bad", "sad", "poor", "hate"]

def simple_nlp(sentence):

    print("Original Sentence:", sentence)

    words = sentence.lower().split()
    print("\nTokens:", words)

    print("\nSyntactic Analysis:")

    for word in words:
        if word in nouns:
            print(word, "→ Noun")
        elif word in verbs:
            print(word, "→ Verb")
        else:
            print(word, "→ Other")

    positive_count = 0
    negative_count = 0

    for word in words:
        if word in positive_words:
            positive_count += 1
        elif word in negative_words:
            negative_count += 1

    print("\nSentiment Analysis:")

    if positive_count > negative_count:
        print("Positive")
    elif negative_count > positive_count:
        print("Negative")
    else:
        print("Neutral")

sentence = "Student study hard for exam but exam is bad"
simple_nlp(sentence)

import nltk
import spacy
from textblob import TextBlob
from nltk.tokenize import word_tokenize
from nltk import pos_tag

nltk.download('punkt')
nltk.download('averaged_perceptron_tagger')

nlp = spacy.load("en_core_web_sm")

def spell_check_nlp(sentence):

    print("\nOriginal Sentence:")
    print(sentence)

    tokens = word_tokenize(sentence)
    print("\nTokens:")
    print(tokens)

    corrected_words = []

    for word in tokens:
        if word.isalpha():
            corrected_word = str(TextBlob(word).correct())
        else:
            corrected_word = word
        corrected_words.append(corrected_word)

    corrected_sentence = " ".join(corrected_words)

    print("\nCorrected Sentence:")
    print(corrected_sentence)

    pos_tags = pos_tag(corrected_words)
    print("\nPOS Tags:")
    print(pos_tags)

    doc = nlp(corrected_sentence)

    print("\nDependencies:")
    for token in doc:
        print(token.text, token.dep_, token.head.text)

    return corrected_sentence


user_input = input("Enter sentence: ")
spell_check_nlp(user_input)
