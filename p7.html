Practical 7: BFS & DFS (Water Jug Problem)

BFS (Water Jug)


from collections import deque

def bfs():
  jugA, jugB = 4, 2
  start = (0, 0)
  goal = (4, 0)

  queue = deque([start])
  visited = set([start])
  parent = {start: None}

  while queue:
    a, b = queue.popleft()

    if (a, b) == goal:
      break

    states = [
        (jugA, b),
        (a, jugB),
        (0, b),
        (a, 0),
        (min(a + b, jugA), max(0, a + b - jugA)),
        (max(0, a + b - jugB), min(a + b, jugB)),
    ]

    for state in states:
      if state not in visited:
        visited.add(state)
        parent[state] = (a, b)
        queue.append(state)

  # print path
  path = []
  cur = goal
  while cur:
    path.append(cur)
    cur = parent[cur]

  print("Path to fill Jug A to 4 liters (BFS):")
  for state in reversed(path):
    print(state)

bfs()

from collections import deque

def dfs():
    jugA, jugB = 4, 2
    start = (0, 0)
    goal = (4, 0)

    stack = deque([start])
    visited = set([start])
    parent = {start: None}

    while stack:
        a, b = stack.pop()

        if (a, b) == goal:
            break

        states = [
            (jugA, b),
            (a, jugB),
            (0, b),
            (a, 0),
            (min(a + b, jugA), max(0, a + b - jugA)),
            (max(0, a + b - jugB), min(a + b, jugB)),
        ]

        for state in states:
            if state not in visited:
                visited.add(state)
                parent[state] = (a, b)
                stack.append(state)

    path = []
    cur = goal
    while cur:
        path.append(cur)
        cur = parent[cur]

    print("Path to fill Jug A to 4 liters (DFS):")
    for state in reversed(path):
        print(state)

dfs()
